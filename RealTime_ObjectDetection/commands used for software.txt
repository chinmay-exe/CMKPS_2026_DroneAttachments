8.Human detection in live video
code:
import cv2
import numpy as np

net = cv2.dnn.readNet("yolov3-tiny.weights", "yolov3-tiny.cfg")

with open("coco.names", "r") as f:
    classes = [line.strip() for line in f.readlines()]

layer_names = net.getLayerNames()
output_layers = [layer_names[i - 1] for i in net.getUnconnectedOutLayers().flatten()]

video = cv2.VideoCapture("video.mp4")  # Replace with your video file

while True:
    ret, frame = video.read()

    if not ret:
        break

    height, width = frame.shape[:2]

    blob = cv2.dnn.blobFromImage(
        frame, 1/255.0, (320, 320),
        swapRB=True, crop=False
    )

    net.setInput(blob)
    outputs = net.forward(output_layers)

    for output in outputs:
        for detection in output:
            scores = detection[5:]
            class_id = np.argmax(scores)
            confidence = scores[class_id]

            if class_id == 0 and confidence > 0.5:  # person
                center_x = int(detection[0] * width)
                center_y = int(detection[1] * height)
                w = int(detection[2] * width)
                h = int(detection[3] * height)

                x = int(center_x - w / 2)
                y = int(center_y - h / 2)

                cv2.rectangle(frame, (x, y), (x+w, y+h),
                              (0, 255, 0), 2)
                cv2.putText(frame, "Person",
                            (x, y-10),
                            cv2.FONT_HERSHEY_SIMPLEX,
                            0.5, (0, 255, 0), 2)

    cv2.imshow("Human Detection", frame)

    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

video.release()
cv2.destroyAllWindows()

cat > ~/droneapp/app.py << 'EOF'
from flask import Flask, render_template, jsonify
import cv2
import numpy as np
import threading
import time

app = Flask(__name__)

person_count = 0
lock = threading.Lock()

net = cv2.dnn.readNet("yolov3-tiny.weights", "yolov3-tiny.cfg")
with open("coco.names", "r") as f:
    classes = [line.strip() for line in f.readlines()]
layer_names = net.getUnconnectedOutLayersNames()

def detect_persons():
    global person_count
    cap = cv2.VideoCapture(0, cv2.CAP_V4L2)
    cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
    cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)

    while True:
        ret, frame = cap.read()
        if not ret:
            time.sleep(0.1)
            continue

        blob = cv2.dnn.blobFromImage(frame, 1/255.0, (416, 416),
                                      swapRB=True, crop=False)
        net.setInput(blob)
        outputs = net.forward(layer_names)

        count = 0
        for output in outputs:
            for detection in output:
                scores = detection[5:]
                class_id = np.argmax(scores)
                confidence = scores[class_id]
                if confidence > 0.5 and classes[class_id] == "person":
                    count += 1

        with lock:
            person_count = count

        time.sleep(0.1)

@app.route('/')
def index():
    return render_template('index.html')

@app.route('/count')
def count():
    with lock:
        return jsonify({"persons": person_count})

if __name__ == '__main__':
    t = threading.Thread(target=detect_persons, daemon=True)
    t.start()
    app.run(host='0.0.0.0', port=5000, debug=False)
EOF

code: (setting up pi 4 as hotspot)
sudo apt install hostapd dnsmasq -y
sudo systemctl stop hostapd dnsmasq
sudo nano /etc/hostapd/hostapd.conf
add the below mentioned code at the bottom: 
(interface=wlan0
driver=nl80211
ssid=DronePI4
hw_mode=g
channel=7
wpa=2
wpa_passphrase=drone1234
wpa_key_mgmt=WPA-PSK
rsn_pairwise=CCMP
)
sudo nano /etc/dnsmasq.conf
add the below mentioned code at the bottom:
(interface=wlan0
dhcp-range=192.168.4.2,192.168.4.20,255.255.255.0,24h)
sudo nano /etc/dhcpcd.conf
add the code mentioned below at the bottom:
(interface=wlan0
static ip_address=192.168.4.1/24
nohook wpa_supplicant)
